## toUpperCase-toLowerCase

```js
// Determinando o tamanho de uma fonte

import React from "react"; 

//toUpperCase, letra maiuscula
let nomeMaiusculo = "BETO"

//toLowerCase, letra maiuscula
let nomeMinusculo = "Beto"

function ElementoJsx() {   
  return (
    <div>
    <br/>
     {nomeMaiusculo.toUpperCase()}
     <br/>
     {nomeMinusculo.toLowerCase()}
    </div>
  );
}

export default ElementoJsx;  
```
